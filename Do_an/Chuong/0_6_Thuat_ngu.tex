\begin{longtable}{l p{11.5cm}}
	\hline
   \textbf{Thuật ngữ}  & \textbf{Giải nghĩa tiếng Việt} \\ \hline 
	\textbf{Actor} & Đối tượng cơ bản có thể đặt hoặc sinh ra trong màn chơi của Unreal Engine \\
	\textbf{Animation} & Hoạt ảnh / Diễn hoạt của các mô hình trong trò chơi \\
	\textbf{Asset} & Tài nguyên phát triển (mô hình 3D, âm thanh, hiệu ứng, v.v.) \\
	\textbf{Behavior Tree} & Cây hành vi (thuật toán điều khiển Trí tuệ nhân tạo) \\
	\textbf{Blueprint} & Hệ thống kịch bản lập trình trực quan của Unreal Engine \\
	\textbf{Blueprint Interface} & Giao diện giao tiếp dữ liệu giữa các Blueprint \\
	\textbf{Bullet Time} & Cơ chế làm chậm thời gian \\
	\textbf{Collider} & Hộp/khối giới hạn dùng để xử lý va chạm vật lý \\
	\textbf{Component} & Thành phần chức năng đính kèm vào một Actor \\
	\textbf{Crosshair} & Tâm ngắm vũ khí trên màn hình \\
    \textbf{Debugging} & Quá trình gỡ lỗi hệ thống \\
    \textbf{Fail-safe} & Cơ chế phòng lỗi, dự phòng khi hệ thống xử lý sai lệch \\
    \textbf{Frame} & Khung hình hiển thị trên màn hình \\
    \textbf{Hit Result} & Dữ liệu trả về khi có va chạm tia quét (vị trí chạm, pháp tuyến) \\
    \textbf{Indie} & Nhà phát triển độc lập (Independent) \\
    \textbf{Level} & Màn chơi / Cấp độ \\
    \textbf{Line Trace} & Thuật toán bắn tia vô hình dò tìm va chạm \\
    \textbf{Lumen} & Hệ thống chiếu sáng toàn cục thời gian thực của Unreal Engine 5 \\
    \textbf{Mechanic} & Cơ chế / Tính năng cốt lõi của trò chơi \\
    \textbf{Mesh} & Lưới đa giác tạo nên hình dáng 3D của vật thể \\
    \textbf{NavMesh} & Lưới điều hướng giúp tính toán đường đi cho AI (Navigation Mesh) \\
    \textbf{Object Pooling} & Kỹ thuật tái sử dụng đối tượng trong bộ nhớ \\
    \textbf{Pawn} & Một dạng Actor có thể bị điều khiển bởi người chơi hoặc AI \\
    \textbf{Portal} & Cổng không gian / Cổng dịch chuyển tức thời \\
    \textbf{Prototype} & Mẫu thử / Nguyên mẫu của sản phẩm \\
    \textbf{Raycast} & Thuật toán chiếu tia tính toán điểm chạm \\
    \textbf{Render Targets} & Kỹ thuật kết xuất hình ảnh camera lên một bề mặt vật lý \\
    \textbf{Respawn} & Tái sinh / Hồi sinh nhân vật \\
    \textbf{Reverse Engineering} & Kỹ thuật dịch ngược, phân tích logic hệ thống nguồn \\
    \textbf{SceneCapture2D} & Thành phần giả lập camera ghi hình môi trường xuất thành kết cấu 2D \\
    \textbf{Spawn} & Quá trình khởi tạo đối tượng mới theo thời gian thực \\
    \textbf{Spawner} & Hệ thống có chức năng tự động tạo ra thực thể mới \\
    \textbf{State Machine} & Máy trạng thái (Một phương pháp lập trình logic cho AI/Hoạt ảnh) \\
    \textbf{Time Dilation} & Thuật toán giãn nở thời gian \\
    \textbf{Trace Complex} & Chế độ kiểm tra va chạm độ chính xác cao dựa trên bề mặt lưới \\
    \textbf{Turret} & Tháp súng tự động \\
    \textbf{Widget} & Thành phần giao diện đồ họa người dùng \\
    \hline
\end{longtable}
